class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        
        #since given matrix is sorted row wisw and column wise, we can apply modified form of binary search
        #take start = 0 and end = len(list[0])-1
        #if target < end : end-- 
        #if target > end : start ++
        
        #idea is that if target is present in thematrix, start and end pointers will be able to capture it else either of them will cross the lenght of the matrix
        
        start = 0
        end = len(matrix[0])-1
        # print(len(matrix[0]),len(matrix))

        while((end>=0) and (start<len(matrix))):
            print(start,end)
            if(target==matrix[start][end]):
                print(start,end)
                return True
            if(target<matrix[start][end]):
                end-=1
            if(target>matrix[start][end]):
                start+=1

            
        print(start,end)

        return False
